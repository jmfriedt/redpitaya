ipaddr=192.168.0.10
serverip=192.168.0.2
gatewayip=192.168.0.2
netmask=255.255.255.0
hostname=redpitaya
modeboot=mmcboot
mmcargs=setenv bootargs console=ttyPS0,115200 root=/dev/mmcblk0p2 rw earlyprintk ip=${ipaddr}:${serverip}:${gatewayip}:${netmask}:${hostname}:eth0:off
mmcboot=run fpgaboot; run mmcargs; fatload mmc 0 0x1000000 ${kernel_image} && fatload mmc 0 0x3000000 ${devicetree_image} && bootm 0x1000000 - 0x3000000
